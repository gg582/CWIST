90static void SHA1Update(SHA1_CTX *context, const uint8_t *data, uint32_t len) {
91 uint32_t i, j;
92 j = context->count[0];
93 if ((context->count[0] += len << 3) < j)
94 context->count[1]++;
95 context->count[1] += (len >> 29);
96 j = (j >> 3) & 63;
97 if ((j + len) > 63) {
98 memcpy(&context->buffer[j], data, (i = 64 - j));
99 SHA1Transform(context->state, context->buffer);
100 for (; i + 63 < len; i += 64) {
101 SHA1Transform(context->state, &data[i]);
102 }
103 j = 0;
104 } else
105 i = 0;
106 memcpy(&context->buffer[j], &data[i], len - i);
107}

110 unsigned i;
111 uint8_t finalcount[8];
112 uint8_t c;
113 for (i = 0; i < 8; i++) {
114 finalcount[i] = (unsigned char)((context->count[(i >= 4 ? 0 : 1)] >> ((3 - (i & 3)) * 8)) & 255);
115 }
116 c = 0200;
117 SHA1Update(context, &c, 1);
118 while ((context->count[0] & 504) != 448) {
119 c = 0000;
120 SHA1Update(context, &c, 1);
121 }
122 SHA1Update(context, finalcount, 8);

148 *output_length = 4 * ((input_length + 2) / 3);
