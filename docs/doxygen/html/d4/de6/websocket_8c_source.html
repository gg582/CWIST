16cwist_websocket *cwist_websocket_upgrade(cwist_http_request *req, int client_fd) {
17 if (!req || client_fd < 0) return NULL;
18
19 // Validate Headers
20 char *connection = cwist_http_header_get(req->headers, "Connection");
21 char *upgrade = cwist_http_header_get(req->headers, "Upgrade");
22 char *key = cwist_http_header_get(req->headers, "Sec-WebSocket-Key");
23
24 if (!connection || !upgrade || !key) return NULL;
25 if (strcasestr(connection, "Upgrade") == NULL) return NULL;
26 if (strcasecmp(upgrade, "websocket") != 0) return NULL;
27
28 // Handshake Key Generation
29 char combined_key[512];
30 snprintf(combined_key, sizeof(combined_key), "%s%s", key, WS_GUID);
31
32 uint8_t hash[20];
33 sha1((uint8_t *)combined_key, strlen(combined_key), hash);
34
35 size_t accept_len;
36 char *accept_key = base64_encode(hash, 20, &accept_len);
37 if (!accept_key) return NULL;
38
39 // Send Response
40 cwist_http_response *res = cwist_http_response_create();
41 res->status_code = 101;
42 cwist_sstring_assign(res->status_text, "Switching Protocols");
43 cwist_http_header_add(&res->headers, "Upgrade", "websocket");
44 cwist_http_header_add(&res->headers, "Connection", "Upgrade");
45 cwist_http_header_add(&res->headers, "Sec-WebSocket-Accept", accept_key);
46
47 cwist_error_t err = cwist_http_send_response(client_fd, res);
48
49 cwist_http_response_destroy(res);
50 cwist_free(accept_key);
51
52 if (err.error.err_i16 != 0) return NULL;
53
54 req->upgraded = true;
55
56 cwist_websocket *ws = (cwist_websocket *)cwist_alloc(sizeof(cwist_websocket));
57 ws->fd = client_fd;
58 ws->is_closed = false;
59 return ws;
60}

62static ssize_t read_exact(int fd, void *buf, size_t len) {
63 size_t total = 0;
64 while (total < len) {

72cwist_ws_frame *cwist_websocket_receive(cwist_websocket *ws) {
73 if (!ws || ws->is_closed) return NULL;
74
75 uint8_t head[2];
76 if (read_exact(ws->fd, head, 2) < 0) return NULL;
77
78 bool fin = (head[0] & 0x80) != 0;
79 cwist_ws_opcode_t opcode = head[0] & 0x0F;
80 bool masked = (head[1] & 0x80) != 0;
81 uint64_t payload_len = head[1] & 0x7F;
82
83 if (!masked) {
84 // Client-to-server frames must be masked per spec
85 // We can choose to strict close or allow. Strict is better.
86 // For now, let's just return error.
87 return NULL;
88 }
89
90 if (payload_len == 126) {
91 uint16_t len16;
92 if (read_exact(ws->fd, &len16, 2) < 0) return NULL;
93 payload_len = ntohs(len16);
94 } else if (payload_len == 127) {
95 uint64_t len64;
96 if (read_exact(ws->fd, &len64, 8) < 0) return NULL;
97 // manually swap if no be64toh
98 // assuming be64toh or similar exists, or manual
99 // Linux usually has be64toh in <endian.h>
100 // Let's implement manual swap to be portable
101 uint8_t *p = (uint8_t *)&len64;
102 payload_len = ((uint64_t)p[0] << 56) | ((uint64_t)p[1] << 48) |
103 ((uint64_t)p[2] << 40) | ((uint64_t)p[3] << 32) |
104 ((uint64_t)p[4] << 24) | ((uint64_t)p[5] << 16) |
105 ((uint64_t)p[6] << 8) | ((uint64_t)p[7]);
106 }
107
108 uint8_t masking_key[4];
109 if (read_exact(ws->fd, masking_key, 4) < 0) return NULL;
110
111 uint8_t *payload = NULL;
112 if (payload_len > 0) {
113 payload = (uint8_t *)cwist_alloc(payload_len + 1); // +1 for safety null term if text
114 if (!payload) return NULL;
115 if (read_exact(ws->fd, payload, payload_len) < 0) {
116 cwist_free(payload);
117 return NULL;
118 }
119
120 // Unmask
121 for (uint64_t i = 0; i < payload_len; i++) {
122 payload[i] ^= masking_key[i % 4];
123 }
124 payload[payload_len] = '\0'; // Null terminate for convenience if text
125 }
126
127 if (opcode == CWIST_WS_FRAME_CLOSE) {
128 ws->is_closed = true;
129 }
130
131 cwist_ws_frame *frame = (cwist_ws_frame *)cwist_alloc(sizeof(cwist_ws_frame));
132 frame->fin = fin;
133 frame->opcode = opcode;
134 frame->payload = payload;
135 frame->payload_len = payload_len;
136
137 return frame;
138}

140int cwist_websocket_send(cwist_websocket *ws, cwist_ws_opcode_t opcode, const uint8_t *data, size_t len) {
141 if (!ws || ws->is_closed) return -1;
142
143 uint8_t head[10]; // Max header size (2 + 8)
144 size_t head_len = 2;
145
146 head[0] = 0x80 | (opcode & 0x0F); // FIN=1
147
148 if (len < 126) {
149 head[1] = len;
150 } else if (len < 65536) {
151 head[1] = 126;
152 head[2] = (len >> 8) & 0xFF;
153 head[3] = len & 0xFF;
154 head_len += 2;
155 } else {
156 head[1] = 127;
157 for (int i = 0; i < 8; i++) {
158 head[2 + i] = (len >> ((7 - i) * 8)) & 0xFF;
159 }
160 head_len += 8;
161 }
162
163 // Server does not mask frames
164
165 if (send(ws->fd, head, head_len, 0) < 0) return -1;
166 if (len > 0) {
167 if (send(ws->fd, data, len, 0) < 0) return -1;
168 }
169
170 return 0;
171}
