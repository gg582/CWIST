25static uint64_t nuke_estimate_required_ram(const char *disk_path) {
26 struct stat st;
27 uint64_t db_size = 0;
28 if (disk_path && stat(disk_path, &st) == 0 && st.st_size > 0) {
29 db_size = (uint64_t)st.st_size;
30 }
31 // Need room for the DB plus SQLite cache/overhead. Double the DB and add base buffer.
32 uint64_t required = (db_size * 2) + CWIST_MIB(32);
33 if (required < CWIST_MIB(64)) {
34 required = CWIST_MIB(64);
35 }
36 return required;
37}

39static bool nuke_disk_has_tables(sqlite3 *db) {
40 if (!db) return true;
41 sqlite3_stmt *stmt = NULL;
42 bool has_tables = true;
43 if (sqlite3_prepare_v2(db,
44 "SELECT count(*) FROM sqlite_master WHERE type='table';",
45 -1,
46 &stmt,
47 NULL) == SQLITE_OK) {
48 if (sqlite3_step(stmt) == SQLITE_ROW) {
49 has_tables = sqlite3_column_int(stmt, 0) > 0;
50 }
51 }
52 if (stmt) sqlite3_finalize(stmt);
53 return has_tables;

56static bool nuke_integrity_ok(sqlite3 *db) {
57 if (!db) return false;
58 sqlite3_stmt *stmt = NULL;
59 bool healthy = false;
60 if (sqlite3_prepare_v2(db, "PRAGMA integrity_check;", -1, &stmt, NULL) == SQLITE_OK) {
61 if (sqlite3_step(stmt) == SQLITE_ROW) {
62 const unsigned char *state = sqlite3_column_text(stmt, 0);
63 healthy = (state && strcmp((const char *)state, "ok") == 0);
64 }
65 }
66 if (stmt) sqlite3_finalize(stmt);
67 return healthy;
68}

70// Helper: Backup source_db to dest_db
71static int nuke_backup(sqlite3 *dest, sqlite3 *source) {
72 if (!dest || !source) return -1;
73 sqlite3_backup *backup = sqlite3_backup_init(dest, "main", source, "main");
74 if (!backup) {
75 fprintf(stderr, "[NukeDB] sqlite3_backup_init failed: %s\n", sqlite3_errmsg(dest));
76 return -1;
77 }
78
79 int rc = sqlite3_backup_step(backup, -1); // Copy all pages
80 if (rc != SQLITE_DONE) {
81 fprintf(stderr, "[NukeDB] Backup step failed: %d (%s)\n", rc, sqlite3_errmsg(dest));
82 }
83
84 sqlite3_backup_finish(backup);
85 return (rc == SQLITE_DONE) ? 0 : -1;
86}

88static void nuke_switch_to_disk(void) {
89 if (g_nuke.is_disk_mode) return;

91 printf("[NukeDB] CRITICAL: Low RAM detected (%zu bytes). Switching to Disk DB.\n", (size_t)cwist_get_available_ram());

116int cwist_nuke_sync(void) {
117 pthread_mutex_lock(&g_nuke_lock);
118
119 if (g_nuke.is_disk_mode) {
120 if (g_nuke.disk_db) {
121 sqlite3_wal_checkpoint_v2(g_nuke.disk_db, "main", SQLITE_CHECKPOINT_PASSIVE, NULL, NULL);
122 pthread_mutex_unlock(&g_nuke_lock);
123 return 0;
124 }
125 pthread_mutex_unlock(&g_nuke_lock);
126 return -1;
127 }
128
129 // Safety: If initial load failed, don't overwrite disk with empty memory DB
130 if (!g_nuke.load_successful) {
131 pthread_mutex_unlock(&g_nuke_lock);
132 return -1;
133 }
134
135 if (!g_nuke.mem_db || !g_nuke.disk_db) {
136 pthread_mutex_unlock(&g_nuke_lock);
137 return -1;
138 }
139
140 int rc = nuke_backup(g_nuke.disk_db, g_nuke.mem_db);
141 pthread_mutex_unlock(&g_nuke_lock);
142 return rc;
143}

166 g_nuke.auto_sync = false;
167 g_nuke.sync_interval_ms = 0;
168 g_nuke.is_disk_mode = false;
169 g_nuke.load_successful = false;
170
171 pthread_mutex_unlock(&g_nuke_lock);
172}

174void cwist_nuke_close(void) {
175 if (!g_running) return;
176 g_running = false;
177
178 // Wake up the sync thread if it's sleeping/waiting
179 if (g_sync_thread != 0) {
180 pthread_kill(g_sync_thread, SIGUSR1);
181 pthread_join(g_sync_thread, NULL);
182 g_sync_thread = 0;
183 }
184
185 cwist_nuke_sync();
186 nuke_cleanup_internal();
187}

189static void *sync_thread_func(void *arg) {
190 CWIST_UNUSED(arg);
191 struct timespec timeout;
192 int signum;
193
194 while (g_running) {
195 timeout.tv_sec = g_nuke.sync_interval_ms / 1000;
196 timeout.tv_nsec = (g_nuke.sync_interval_ms % 1000) * 1000000;
197
198 // Wait for signals (INT, TERM, USR1, USR2) or Timeout
199 signum = sigtimedwait(&g_sigset, NULL, &timeout);
200
201 if (signum > 0) {
202 if (signum == SIGUSR1) {
203 // Requested to stop via cwist_nuke_close
204 break;
205 }
206
207 if (signum == SIGUSR2) {
208 // Immediate sync requested via commit hook
209 cwist_nuke_sync();
210 continue;
211 }
212
213 // Handle termination signals (SIGINT, SIGTERM)
214 printf("\n[NukeDB] Intercepted Signal %d. Saving data...\n", signum);
215
216 cwist_nuke_sync();
217 nuke_cleanup_internal();
218 g_running = false;
219
220 // Unblock the signal first
221 sigset_t s;
222 sigemptyset(&s);
223 sigaddset(&s, signum);
224 pthread_sigmask(SIG_UNBLOCK, &s, NULL);
225
226 // Restore default handler and re-raise
227 signal(signum, SIG_DFL);
228 raise(signum);
229 return NULL;
230 } else { // Timeout (EAGAIN) or Interruption
231 if (errno == EAGAIN && g_running && g_nuke.auto_sync) {
232 if (cwist_is_ram_critical(CWIST_MIB(128))) {
233 nuke_switch_to_disk();
234 } else {
235 cwist_nuke_sync();
236 }
237 }
238 }
239 }
240 return NULL;
241}

243int cwist_nuke_init(const char *disk_path, int sync_interval_ms) {
244 if (g_running) return CWIST_NUKE_ERR_GENERIC; // Already running
245 if (!disk_path) return CWIST_NUKE_ERR_GENERIC;
246
247 uint64_t required_ram = nuke_estimate_required_ram(disk_path);
248 uint64_t available_ram = cwist_get_available_ram();
249 if (available_ram > 0 && available_ram < required_ram) {
250 fprintf(stderr,
251 "[NukeDB] Insufficient RAM for in-memory mode (have %zu bytes, need ~%zu bytes).\n",
252 (size_t)available_ram,
253 (size_t)required_ram);
254 return CWIST_NUKE_ERR_LOW_MEMORY;
255 }
256
257 g_nuke.disk_path = cwist_strdup(disk_path);
258 if (!g_nuke.disk_path) {
259 return CWIST_NUKE_ERR_GENERIC;
260 }
261 g_nuke.sync_interval_ms = sync_interval_ms > 0 ? sync_interval_ms : 1000;
262 g_nuke.auto_sync = (sync_interval_ms > 0);
263 g_nuke.is_disk_mode = false;
264 g_nuke.load_successful = false;
265
266 // 1. Open Disk DB
267 int rc = sqlite3_open(disk_path, &g_nuke.disk_db);
268 if (rc != SQLITE_OK) {
269 fprintf(stderr, "[NukeDB] Failed to open disk DB: %s\n", sqlite3_errmsg(g_nuke.disk_db));
270 nuke_cleanup_internal();
271 return CWIST_NUKE_ERR_GENERIC;
272 }
273
274 if (!nuke_integrity_ok(g_nuke.disk_db)) {
275 fprintf(stderr, "[NukeDB] Integrity check failed for '%s'. Aborting in-memory mode.\n", disk_path);
276 nuke_cleanup_internal();
277 return CWIST_NUKE_ERR_GENERIC;
278 }
279
280 sqlite3_exec(g_nuke.disk_db, "PRAGMA journal_mode=WAL;", NULL, NULL, NULL);
281 sqlite3_exec(g_nuke.disk_db, "PRAGMA synchronous=NORMAL;", NULL, NULL, NULL);
282
283 // 2. Open Memory DB
284 rc = sqlite3_open(":memory:", &g_nuke.mem_db);
285 if (rc != SQLITE_OK) {
286 fprintf(stderr, "[NukeDB] Failed to open memory DB: %s\n", sqlite3_errmsg(g_nuke.mem_db));
287 nuke_cleanup_internal();
288 return CWIST_NUKE_ERR_GENERIC;
289 }
290
291 // 3. Load Disk -> Memory
292 pthread_mutex_lock(&g_nuke_lock);
293
294 // Use sqlite3_backup as it is safer than raw deserialize for WAL/mode transitions
295 int backup_rc = nuke_backup(g_nuke.mem_db, g_nuke.disk_db);
296 if (backup_rc == 0) {
297 g_nuke.load_successful = true;
298 } else if (!nuke_disk_has_tables(g_nuke.disk_db)) {
299 g_nuke.load_successful = true;
300 } else {
301 fprintf(stderr, "[NukeDB] Initial disk->memory load failed. Falling back to disk-only mode.\n");
302 g_nuke.is_disk_mode = true;
303 sqlite3_close(g_nuke.mem_db);
304 g_nuke.mem_db = NULL;
305 }
306 pthread_mutex_unlock(&g_nuke_lock);
307
308 // Register commit hook for immediate sync. Only needed while in-memory mode is active.
309 if (g_nuke.mem_db && !g_nuke.is_disk_mode) {
310 sqlite3_commit_hook(g_nuke.mem_db, nuke_commit_hook, NULL);
311 }
312
313 g_running = true;
314
315 // 4. Setup Signal Interception
316 sigemptyset(&g_sigset);
317 sigaddset(&g_sigset, SIGINT);
318 sigaddset(&g_sigset, SIGTERM);
319 sigaddset(&g_sigset, SIGUSR1);
320 sigaddset(&g_sigset, SIGUSR2);
321
322 pthread_sigmask(SIG_BLOCK, &g_sigset, NULL);
323
324 // 5. Start Sync/Signal Thread
325 if (pthread_create(&g_sync_thread, NULL, sync_thread_func, NULL) != 0) {
326 g_running = false;
327 pthread_sigmask(SIG_UNBLOCK, &g_sigset, NULL);
328 nuke_cleanup_internal();
329 return CWIST_NUKE_ERR_GENERIC;
330 }
331
332 // 6. Register atexit for ultimate safety
333 atexit(cwist_nuke_close);
334
335 return CWIST_NUKE_OK;
336}
