48static bool bdr_entry_should_decay(const cwist_bdr_t *bdr, const bdr_entry_t *entry, time_t now) {
49 if (!bdr || !entry) return false;
50 if (bdr->max_entry_age_sec > 0 && entry->created_at > 0) {
51 if (now - entry->created_at > bdr->max_entry_age_sec) {
52 return true;
53 }
54 }
55 if (entry->is_stable && bdr->revalidate_hits > 0 && entry->hits >= bdr->revalidate_hits) {
56 return true;
57 }
58 return false;

61static bool bdr_trim_oldest(cwist_bdr_t *bdr, time_t now) {
62 if (!bdr) return false;
63 size_t victim_idx = SIZE_MAX;
64 bdr_entry_t *victim = NULL;
65 bdr_entry_t *victim_prev = NULL;
66 time_t oldest = now;
67
68 for (size_t i = 0; i < bdr->bucket_count; ++i) {
69 bdr_entry_t *prev = NULL;
70 bdr_entry_t *curr = bdr->buckets[i];
71 while (curr) {
72 if (curr->response_blob && (!victim || curr->created_at < oldest)) {
73 victim = curr;
74 victim_prev = prev;
75 victim_idx = i;
76 oldest = curr->created_at;
77 }
78 prev = curr;
79 curr = curr->next;
80 }
81 }
82
83 if (!victim || victim_idx == SIZE_MAX) return false;
84 bdr_remove_entry(bdr, victim_idx, victim_prev, victim);
85 return true;
86}

88static void bdr_sweep(cwist_bdr_t *bdr, size_t steps) {
89 if (!bdr || bdr->bucket_count == 0 || steps == 0) return;
90 time_t now = time(NULL);
91 for (size_t i = 0; i < steps; ++i) {
92 size_t idx = bdr->gc_cursor % bdr->bucket_count;
93 bdr->gc_cursor = (bdr->gc_cursor + 1) % bdr->bucket_count;
94 bdr_entry_t *prev = NULL;
95 bdr_entry_t *curr = bdr->buckets[idx];
96 while (curr) {
97 if (bdr_entry_should_decay(bdr, curr, now)) {
98 bdr_entry_t *victim = curr;
99 curr = curr->next;
100 bdr_remove_entry(bdr, idx, prev, victim);
101 continue;
102 }
103 prev = curr;
104 curr = curr->next;
105 }
106 }

109static void bdr_guardrails(cwist_bdr_t *bdr) {
110 if (!bdr) return;
111 bdr_sweep(bdr, BDR_GC_SWEEP);
112
113 if (bdr->max_bytes == 0) return;
114 time_t now = time(NULL);
115 while (bdr->current_bytes > bdr->max_bytes) {
116 if (!bdr_trim_oldest(bdr, now)) break;
117 }

161static void bdr_check_ram(cwist_bdr_t *bdr) {
162 if (bdr->is_disk_mode) return;
163
164 // Threshold: 64MB free (conservative)
165 if (cwist_is_ram_critical(CWIST_MIB(64))) {
166 printf("[BDR] Low RAM. Switching to Disk Cache.\n");
167 // Open Disk DB
168 if (sqlite3_open("cwist_bdr_fallback.db", &bdr->disk_db) == SQLITE_OK) {
169 char *err = NULL;
170 sqlite3_exec(bdr->disk_db, "CREATE TABLE IF NOT EXISTS bdr (hash INTEGER PRIMARY KEY, blob BLOB);", NULL, NULL, &err);
171 if (err) sqlite3_free(err);
172
173 // Move existing memory items to disk
174 sqlite3_exec(bdr->disk_db, "BEGIN TRANSACTION;", NULL, NULL, NULL);
175 for (size_t i = 0; i < bdr->bucket_count; i++) {
176 bdr_entry_t *curr = bdr->buckets[i];
177 while (curr) {
178 if (curr->is_stable) { // Only move stable items
179 sqlite3_stmt *stmt;
180 sqlite3_prepare_v2(bdr->disk_db, "INSERT INTO bdr (hash, blob) VALUES (?, ?);", -1, &stmt, NULL);
181 sqlite3_bind_int64(stmt, 1, curr->request_hash);
182 sqlite3_bind_blob(stmt, 2, curr->response_blob, curr->len, SQLITE_STATIC);
183 sqlite3_step(stmt);
184 sqlite3_finalize(stmt);
185 }
186
187 // Free memory
188 bdr_entry_t *next = curr->next;
189 bdr_release_blob(bdr, curr);
190 cwist_free(curr);
191 curr = next;
192 }
193 bdr->buckets[i] = NULL;
194 }
195 sqlite3_exec(bdr->disk_db, "COMMIT;", NULL, NULL, NULL);
196 bdr->is_disk_mode = true;
197 }
