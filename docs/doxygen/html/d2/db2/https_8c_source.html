70 }
71
72 *ctx = (cwist_https_context*)cwist_alloc(sizeof(cwist_https_context));
73 if (!*ctx) {
74 SSL_CTX_free(ssl_ctx);
75 err.error.err_i16 = -1;

97 cwist_error_t err = make_error(CWIST_ERR_INT16);
98
99 if (!ctx || !ctx->ctx || client_fd < 0) {
100 err.error.err_i16 = -1;
101 return err;
102 }
103
104 SSL *ssl = SSL_new(ctx->ctx);
105 if (!ssl) {
106 return make_ssl_error("Failed to create SSL structure");
107 }
108
109 SSL_set_fd(ssl, client_fd);
110
111 if (SSL_accept(ssl) <= 0) {
112 // Handshake failed
113 // We capture the error before freeing
114 cwist_error_t ssl_err = make_ssl_error("SSL handshake failed");
115 SSL_free(ssl);
116 return ssl_err;
117 }
118
119 *conn = (cwist_https_connection*)cwist_alloc(sizeof(cwist_https_connection));
120 if (!*conn) {
121 SSL_free(ssl);
122 err.error.err_i16 = -1;
123 return err;
124 }
125
126 (*conn)->fd = client_fd;
127 (*conn)->ssl = ssl;
128 (*conn)->read_buf = cwist_alloc(CWIST_HTTP_READ_BUFFER_SIZE);
129 if (!(*conn)->read_buf) {
130 SSL_free(ssl);
131 cwist_free(*conn);
132 *conn = NULL;
133 err.error.err_i16 = -1;
134 return err;
135 }
136 (*conn)->buf_len = 0;
137 (*conn)->read_buf[0] = '\0';
138
139 err.error.err_i16 = 0;
140 return err;

163 char *header_end = NULL;
164
165 while (!(header_end = strstr(conn->read_buf, "\r\n\r\n"))) {

167 return NULL;
168 }
169

171 int pret = poll(&pfd, 1, CWIST_HTTP_TIMEOUT_MS);
172 if (pret <= 0) {

174 }

177 if (bytes <= 0) {

179 if (ssl_err == SSL_ERROR_WANT_READ || ssl_err == SSL_ERROR_WANT_WRITE) {
180 continue;

190 if (!req) return NULL;

193

195 size_t body_received = total_received - header_len;
196
197 if (req->content_length > 0) {
198 if (req->content_length > CWIST_HTTP_MAX_BODY_SIZE) {
199 cwist_http_request_destroy(req);
200 return NULL;
201 }
202
203 char *body = cwist_alloc(req->content_length + 1);
204 if (!body) {
205 cwist_http_request_destroy(req);
206 return NULL;
207 }
208
209 size_t to_copy = body_received < req->content_length ? body_received : req->content_length;
210 memcpy(body, header_end + 4, to_copy);
211 size_t current_body_len = to_copy;
212
213 while (current_body_len < req->content_length) {
214 struct pollfd pfd = { .fd = conn->fd, .events = POLLIN };
215 int pret = poll(&pfd, 1, CWIST_HTTP_TIMEOUT_MS);
216 if (pret <= 0) {
217 cwist_free(body);
218 cwist_http_request_destroy(req);
219 return NULL;
220 }
221
222 int bytes = SSL_read(conn->ssl, body + current_body_len, (int)(req->content_length - current_body_len));
223 if (bytes <= 0) {
224 int ssl_err = SSL_get_error(conn->ssl, bytes);
225 if (ssl_err == SSL_ERROR_WANT_READ || ssl_err == SSL_ERROR_WANT_WRITE) {
226 continue;
227 }
228 cwist_free(body);
229 cwist_http_request_destroy(req);
230 return NULL;
231 }
232 current_body_len += (size_t)bytes;
233 }
234 body[req->content_length] = '\0';
235 cwist_sstring_assign_len(req->body, body, req->content_length);
236 cwist_free(body);
237
238 if (body_received > req->content_length) {
239 size_t leftover_len = body_received - req->content_length;
240 memmove(conn->read_buf, header_end + 4 + req->content_length, leftover_len);
241 conn->buf_len = leftover_len;
242 } else {
243 conn->buf_len = 0;
244 }
245 } else {
246 if (body_received > 0) {
247 memmove(conn->read_buf, header_end + 4, body_received);
248 conn->buf_len = body_received;
249 } else {
250 conn->buf_len = 0;
251 }
252 }
253 conn->read_buf[conn->buf_len] = '\0';
254
255 return req;
256}

258cwist_error_t cwist_https_send_response(cwist_https_connection *conn, cwist_http_response *res) {
259 cwist_error_t err = make_error(CWIST_ERR_INT16);
260
261 if (!conn || !conn->ssl || !res) {
262 err.error.err_i16 = -1;
263 return err;
264 }
265
266 // 1. Serialize using existing HTTP logic
267 cwist_sstring *response_str = cwist_http_stringify_response(res);
268 if (!response_str) {
269 err.error.err_i16 = -1;
270 return err;
271 }
272
273 // 2. Send over SSL
274 const char *p = response_str->data;
275 int left = (int)response_str->size;
276 int total_sent = 0;
277
278 err.error.err_i16 = 0; // Assume success initially
279
280 while (left > 0) {
281 int sent = SSL_write(conn->ssl, p, left);
282 if (sent <= 0) {
283 int ssl_err = SSL_get_error(conn->ssl, sent);
284 if (ssl_err == SSL_ERROR_WANT_WRITE || ssl_err == SSL_ERROR_WANT_READ) {
285 continue; // Retry
286 }
287 err = make_ssl_error("SSL write failed");
288 break;
289 }
290 p += sent;
291 left -= sent;
292 total_sent += sent;
293 }
294
295 cwist_sstring_destroy(response_str);
296 return err;
297}
