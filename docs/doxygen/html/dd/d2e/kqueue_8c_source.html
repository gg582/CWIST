48 struct kevent kev;
49 // We use EVFILT_USER to trigger a custom event.

51 EV_SET(&kev, (uintptr_t)job, EVFILT_USER, EV_ADD | EV_ENABLE | EV_ONESHOT, NOTE_TRIGGER, 0, job);
52
53 if (kevent(q->kq_fd, &kev, 1, NULL, 0, NULL) < 0) {

61void cwist_io_queue_run(cwist_io_queue *q) {
