17static pthread_mutex_t rid_mutex = PTHREAD_MUTEX_INITIALIZER;
18static unsigned int rid_seed = 0;

20static char *generate_request_id() {
21 static const char charset[] = "abcdefghijklmnopqrstuvwxyz0123456789";
22 char *id = cwist_alloc(17);
23
24 pthread_mutex_lock(&rid_mutex);
25 if (rid_seed == 0) rid_seed = (unsigned int)time(NULL) ^ (unsigned int)pthread_self();
26 unsigned int seed = rid_seed++;
27 pthread_mutex_unlock(&rid_mutex);
28
29 for (int i = 0; i < 16; i++) {
30 id[i] = charset[rand_r(&seed) % (sizeof(charset) - 1)];
31 }
32 id[16] = '\0';
33 return id;
34}

94typedef struct {
95 char ip[46];
96 long last_reset;
97 int count;
98} ip_limit_t;

100#define MAX_IP_TRACK 1024
101static ip_limit_t ip_cache[MAX_IP_TRACK];
102static int ip_cache_count = 0;
103static pthread_mutex_t rate_mutex = PTHREAD_MUTEX_INITIALIZER;
104
105void cwist_mw_rate_limit_ip_handler(cwist_http_request *req, cwist_http_response *res, cwist_handler_func next) {
106
107 // Get client ip from fd
108 cwist_sstring *ip = cwist_get_client_ip_from_fd(req->client_fd);
109
110 time_t now = time(NULL);
111 ip_limit_t *found = NULL;
112
113 pthread_mutex_lock(&rate_mutex);
114
115 // if ip is found on ip_cache, set found = &ip_cache[i];
116 for (int i = 0; i < ip_cache_count; i++) {
117 if (!cwist_sstring_compare(ip, ip_cache[i].ip)) {
118 found = &ip_cache[i];
119 break;
120 }
121 }
122
123 // if not found, add current ip to ip_cache
124 if (!found && ip_cache_count < MAX_IP_TRACK) {
125 found = &ip_cache[ip_cache_count++];
126 strncpy(found->ip, ip->data, sizeof(found->ip) - 1);
127 found->last_reset = now;
128 found->count = 0;
129 }
130
131 // if found, refresh reset time
132 if (found) {
133 if (now - found->last_reset >= 60) {
134 found->last_reset = now;
135 found->count = 0;
136 }
137
138 // if cound count is more than 60, block request
139 if (found->count >= 60) {
140 pthread_mutex_unlock(&rate_mutex);
141 res->status_code = 429;
142 cwist_sstring_assign(res->body, "Too Many Requests");
143 cwist_http_header_add(&res->headers, "Retry-After", "60");
144 return;
145 }
146 found->count++;
147 }
148 cwist_sstring_destroy(ip);
149 pthread_mutex_unlock(&rate_mutex);
150
151 next(req, res);
152}
